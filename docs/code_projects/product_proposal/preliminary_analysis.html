X = df[['trip_distance_km', 'trip_duration_min', 'service_fee_to_app', 'passenger_count']]

X = sm.add_constant(X)
model = sm.OLS(y, X).fit()

print(model.summary())
#R-squared = 21.3% This means that only 21.3% of the variation in fare_amount_euro (the price of the ride) is explained by the independent variables (distance, duration, driver rating, service fee, and passenger count).
#Adjusted r-squared = This is similar to R-squared but adjusted for the number of predictors. It suggests the model does not improve significantly with more variables.
#Skew = 1.424, Kurtosis = 6.470 â†’ Suggests the model has some skewed residuals and heavy tails, meaning extreme values may be affecting predictions.
#The model as a whole is statistically significant (since the p-value is extremely low), meaning at least one predictor significantly affects fare_amount_euro.

# Remove outliers based on IQR for 'fare_amount_euro' from OLS test
Q1 = df['fare_amount_euro'].quantile(0.25)
Q3 = df['fare_amount_euro'].quantile(0.75)
IQR = Q3 - Q1

# Filter out the rows that are outliers
df_outliers_removed = df[(df['fare_amount_euro'] >= (Q1 - 1.5 * IQR)) & (df['fare_amount_euro'] <= (Q3 + 1.5 * IQR))]
# OLS Regression
X = df_outliers_removed[['trip_duration_min','driver_rating', 'service_fee_to_app', 'passenger_count', 'fuel_cost_per_km']]
y = df_outliers_removed['fare_amount_euro']
X = sm.add_constant(X)
model = sm.OLS(y, X).fit()

print(model.summary())
#Too many influential data points were removed, making the remaining data too uniform, with removal of outliers,  coefficients for other variables are extremely close to zero, suggests that the model isn't really using those variables to predict fare
